declare
k emp.empno%TYPE;
theJob emp.job%TYPE;
theSal emp.sal%TYPE;
begin
     k:=&num;
     select job into theJob from emp where empno=k;
    select sal into theSal from emp where empno=k;
if theJob= 'CLERK'then
    update emp set sal=theSal*1.05 where empno=k;
elsif theJob= 'MANAGER' then
    update emp set sal=theSal*1.1 where empno=k;
else
    update emp set sal=theSal*1.08 where empno=k;
end if;
EXCEPTION
    when NO_DATA_FOUND then
        dbms_output.put_line('警告：没有这个员工！');
end;
