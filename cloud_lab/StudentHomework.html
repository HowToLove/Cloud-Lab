    declare
    k emp.empno%TYPE;
    empRecord emp%ROWTYPE;
    begin
         k:=&num;
         select * into empRecord from emp where empno=k;
    dbms_output.put_line('empno:'||empRecord.empno||'ename:'||empRecord.ename||'job:'||empRecord.job||'mgr:'||empRecord.mgr||'hiredate:'||empRecord.hiredate||'sal:'||empRecord.sal||'comm:'||empRecord.comm||'deptno:'||empRecord.deptno);
    EXCEPTION
        when NO_DATA_FOUND then
            dbms_output.put_line('警告：没有这个员工！');
    end;

    declare
    k emp.empno%TYPE;
    theJob emp.job%TYPE;
    theSal emp.sal%TYPE;
    begin
         k:=&num;
         select job into theJob from emp where empno=k;
        select sal into theSal from emp where empno=k;
    if theJob= 'CLERK'then
        update emp set sal=theSal*1.05 where empno=k;
    elsif theJob= 'MANAGER' then
        update emp set sal=theSal*1.1 where empno=k;
    else
        update emp set sal=theSal*1.08 where empno=k;
    end if;

    declare
        TYPE sample_array IS TABLE OF VARCHAR2(1) INDEX BY BINARY_INTEGER;
        theString varchar2(50);
        temp sample_array;
        reverseString varchar2(50);
        i integer;
    begin
    theString:= '&myString';
    dbms_output.put_line(theString);
    i:= length(theString);
    while i>=1
    loop
        temp(i):=substr(theString,i,1);
        reverseString:=reverseString||temp(i);
        i:=i-1;
    end loop;
    dbms_output.put_line(reverseString);
    end;

    TYPE comm_array IS TABLE OF emp.comm%TYPE INDEX BY BINARY_INTEGER;
    TYPE empno_array IS TABLE OF emp.empno%TYPE INDEX BY BINARY_INTEGER;
    i integer:=1;
    j integer:=1;
    empno empno_array;
        comm comm_array;
        CURSOR mycur(myJob emp.job%TYPE) is select empno,comm from emp where job=myJob;
    begin
        if mycur%ISOPEN=false then
            open mycur('SALESMAN');
        end if;
        fetch mycur into empno(i),comm(i);
        dbms_output.put_line('SALESMAN的职工号及原来佣金');
        while mycur%FOUND
        loop
            dbms_output.put_line(empno(i)|| '<===>'||comm(i));
            i:=i+1;
    fetch mycur into empno(i),comm(i);
        end loop;
        close mycur;
        while j<i
        loop
            update emp set comm=comm(j)+500 where empno=empno(j);
            j:=j+1;
        end loop;
        dbms_output.put_line('SALESMAN的职工号及改后佣金');
    if mycur%ISOPEN=false then
            open mycur('SALESMAN');
        end if;
